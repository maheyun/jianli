# 项目四：产品组合分析方案

## 1. 项目概述

本项目旨在通过整合各平台销售数据，建立新品健康度实时看板与补货预警机制，实现数据驱动的库存管理优化，提高新品售罄率和补货及时率。

## 2. 数据获取

### 数据库结构

#### `products`表
| 字段名 | 数据类型 | 描述 |
|-------|---------|------|
| `product_id` | INT | 产品ID |
| `product_name` | VARCHAR(255) | 产品名称 |
| `category_id` | INT | 品类ID |
| `price` | DECIMAL(10,2) | 价格 |
| `cost` | DECIMAL(10,2) | 成本 |
| `launch_date` | DATE | 上市日期 |

#### `sales_data`表
| 字段名 | 数据类型 | 描述 |
|-------|---------|------|
| `id` | INT | 销售数据ID |
| `product_id` | INT | 产品ID |
| `platform` | VARCHAR(50) | 平台名称 |
| `sale_date` | DATE | 销售日期 |
| `sales_volume` | INT | 销量 |
| `sales_amount` | DECIMAL(10,2) | 销售额 |

#### `inventory_data`表
| 字段名 | 数据类型 | 描述 |
|-------|---------|------|
| `product_id` | INT | 产品ID |
| `current_stock` | INT | 当前库存 |
| `safety_stock` | INT | 安全库存 |
| `lead_time_days` | INT | 补货 lead time（天） |

### MySQL查询语句

```sql
-- 1. 各平台新品销售数据
SELECT 
    p.product_id,
    p.product_name,
    p.category_id,
    p.price,
    p.cost,
    s.platform,
    SUM(s.sales_volume) as sales_volume,
    SUM(s.sales_amount) as sales_amount,
    SUM(s.sales_amount - p.cost * s.sales_volume) as gross_profit
FROM 
    products p
JOIN 
    sales_data s ON p.product_id = s.product_id
WHERE 
    p.launch_date BETWEEN DATE_SUB(NOW(), INTERVAL 30 DAY) AND NOW() -- 新品定义：30天内上市
GROUP BY 
    p.product_id, p.product_name, p.category_id, p.price, p.cost, s.platform;

-- 2. 库存数据
SELECT 
    product_id,
    current_stock,
    safety_stock,
    lead_time_days
FROM 
    inventory_data;

-- 3. 历史销售趋势
SELECT 
    product_id,
    date(sale_date) as sale_date,
    SUM(sales_volume) as daily_sales
FROM 
    sales_data
WHERE 
    sale_date BETWEEN DATE_SUB(NOW(), INTERVAL 30 DAY) AND NOW()
GROUP BY 
    product_id, date(sale_date);
```

## 3. 数据分析

### Python代码

```python
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
import pymysql

# 连接数据库
conn = pymysql.connect(
    host='localhost',
    user='root',
    password='password',
    database='ecommerce'
)

# 1. 各平台新品销售数据
sales_data_query = """
SELECT 
    p.product_id,
    p.product_name,
    p.category_id,
    p.price,
    p.cost,
    s.platform,
    SUM(s.sales_volume) as sales_volume,
    SUM(s.sales_amount) as sales_amount,
    SUM(s.sales_amount - p.cost * s.sales_volume) as gross_profit
FROM 
    products p
JOIN 
    sales_data s ON p.product_id = s.product_id
WHERE 
    p.launch_date BETWEEN DATE_SUB(NOW(), INTERVAL 30 DAY) AND NOW()
GROUP BY 
    p.product_id, p.product_name, p.category_id, p.price, p.cost, s.platform;
"""
sales_data_df = pd.read_sql(sales_data_query, conn)

# 2. 库存数据
inventory_query = """
SELECT 
    product_id,
    current_stock,
    safety_stock,
    lead_time_days
FROM 
    inventory_data;
"""
inventory_df = pd.read_sql(inventory_query, conn)

# 3. 历史销售趋势
sales_trend_query = """
SELECT 
    product_id,
    date(sale_date) as sale_date,
    SUM(sales_volume) as daily_sales
FROM 
    sales_data
WHERE 
    sale_date BETWEEN DATE_SUB(NOW(), INTERVAL 30 DAY) AND NOW()
GROUP BY 
    product_id, date(sale_date);
"""
sales_trend_df = pd.read_sql(sales_trend_query, conn)

# 计算新品健康度指标
product_metrics = sales_data_df.groupby(['product_id', 'product_name', 'category_id', 'price', 'cost']).agg({
    'sales_volume': 'sum',
    'sales_amount': 'sum',
    'gross_profit': 'sum'
}).reset_index()

# 计算毛利率
product_metrics['gross_margin'] = product_metrics['gross_profit'] / product_metrics['sales_amount']

# 计算预计售罄率（假设30天周期）
product_metrics['estimated_sell_through_rate'] = product_metrics['sales_volume'] / (product_metrics['sales_volume'] + inventory_df.set_index('product_id').loc[product_metrics['product_id'], 'current_stock'].values)

# 合并库存数据
product_metrics = pd.merge(product_metrics, inventory_df, on='product_id', how='left')

# 计算补货点和补货量
product_metrics['avg_daily_sales'] = product_metrics['sales_volume'] / 30  # 假设30天销售数据
product_metrics['reorder_point'] = product_metrics['avg_daily_sales'] * product_metrics['lead_time_days'] + product_metrics['safety_stock']
product_metrics['suggested_reorder_quantity'] = product_metrics['avg_daily_sales'] * (product_metrics['lead_time_days'] + 7)  # 7天安全库存

# 产品分级
def product_segment(row):
    if row['estimated_sell_through_rate'] > 0.6 and row['gross_margin'] > 0.4:
        return '爆款'
    elif row['estimated_sell_through_rate'] > 0.4 and row['gross_margin'] > 0.3:
        return '畅销款'
    elif row['estimated_sell_through_rate'] > 0.2:
        return '一般款'
    else:
        return '滞销款'

product_metrics['segment'] = product_metrics.apply(product_segment, axis=1)

# 分析各平台销售情况
platform_analysis = sales_data_df.groupby('platform').agg({
    'sales_volume': 'sum',
    'sales_amount': 'sum',
    'gross_profit': 'sum'
}).reset_index()
platform_analysis['gross_margin'] = platform_analysis['gross_profit'] / platform_analysis['sales_amount']

# 数据可视化
plt.figure(figsize=(15, 12))

# 1. 产品分级分布
plt.subplot(3, 2, 1)
segment_counts = product_metrics['segment'].value_counts()
plt.pie(segment_counts, labels=segment_counts.index, autopct='%1.1f%%')
plt.title('产品分级分布')

# 2. 各平台销售情况
plt.subplot(3, 2, 2)
plt.bar(platform_analysis['platform'], platform_analysis['sales_amount'])
plt.title('各平台销售金额')
plt.ylabel('销售金额')

# 3. 产品健康度散点图
plt.subplot(3, 2, 3)
plt.scatter(product_metrics['estimated_sell_through_rate'], product_metrics['gross_margin'], s=100)
for i, product in enumerate(product_metrics['product_name']):
    plt.annotate(product, (product_metrics['estimated_sell_through_rate'][i], product_metrics['gross_margin'][i]))
plt.title('产品健康度分析：售罄率 vs 毛利率')
plt.xlabel('预计售罄率')
plt.ylabel('毛利率')

# 4. 滞销品分析
plt.subplot(3, 2, 4)
滞销品 = product_metrics[product_metrics['segment'] == '滞销款']
plt.bar(滞销品['product_name'], 滞销品['estimated_sell_through_rate'])
plt.title('滞销品售罄率')
plt.ylabel('售罄率')
plt.xticks(rotation=45, ha='right')

# 5. 补货建议
plt.subplot(3, 2, 5)
补货建议 = product_metrics[product_metrics['current_stock'] < product_metrics['reorder_point']]
plt.bar(补货建议['product_name'], 补货建议['suggested_reorder_quantity'])
plt.title('补货建议')
plt.ylabel('建议补货量')
plt.xticks(rotation=45, ha='right')

# 6. 销售趋势
plt.subplot(3, 2, 6)
# 选择前5个销量最高的产品
top_products = product_metrics.nlargest(5, 'sales_volume')['product_id']
top_sales_trend = sales_trend_df[sales_trend_df['product_id'].isin(top_products)]
for product_id in top_products:
    product_trend = top_sales_trend[top_sales_trend['product_id'] == product_id]
    plt.plot(product_trend['sale_date'], product_trend['daily_sales'], label=f'产品{product_id}')
plt.title('Top 5产品销售趋势')
plt.xlabel('日期')
plt.ylabel('日销量')
plt.legend()

plt.tight_layout()
plt.savefig('产品组合分析.png')

# 关闭数据库连接
conn.close()

# 生成分析报告
with open('产品组合分析报告.md', 'w', encoding='utf-8') as f:
    f.write('# 产品组合分析报告\n\n')
    f.write('## 1. 新品概览\n')
    f.write(f'### 新品总数: {len(product_metrics)}\n')
    f.write(f'### 总销量: {product_metrics["sales_volume"].sum()}\n')
    f.write(f'### 总销售额: {product_metrics["sales_amount"].sum():.2f}\n')
    f.write(f'### 总毛利: {product_metrics["gross_profit"].sum():.2f}\n')
    f.write(f'### 平均毛利率: {product_metrics["gross_margin"].mean():.2%}\n\n')
    
    f.write('## 2. 产品分级分析\n')
    f.write('### 分级分布:\n')
    for segment, count in segment_counts.items():
        f.write(f'- {segment}: {count}个 ({count/len(product_metrics):.2%})\n')
    
    f.write('\n### 分级详情:\n')
    for segment in ['爆款', '畅销款', '一般款', '滞销款']:
        segment_products = product_metrics[product_metrics['segment'] == segment]
        if len(segment_products) > 0:
            f.write(f'#### {segment}:\n')
            f.write(f'- 数量: {len(segment_products)}\n')
            f.write(f'- 平均销量: {segment_products["sales_volume"].mean():.2f}\n')
            f.write(f'- 平均销售额: {segment_products["sales_amount"].mean():.2f}\n')
            f.write(f'- 平均毛利率: {segment_products["gross_margin"].mean():.2%}\n')
            f.write(f'- 平均售罄率: {segment_products["estimated_sell_through_rate"].mean():.2%}\n\n')
    
    f.write('## 3. 平台分析\n')
    for index, row in platform_analysis.iterrows():
        f.write(f'### {row["platform"]}:\n')
        f.write(f'- 销量: {row["sales_volume"]}\n')
        f.write(f'- 销售额: {row["sales_amount"]:.2f}\n')
        f.write(f'- 毛利: {row["gross_profit"]:.2f}\n')
        f.write(f'- 毛利率: {row["gross_margin"]:.2%}\n\n')
    
    f.write('## 4. 补货建议\n')
    if len(补货建议) > 0:
        f.write('### 需要补货的产品:\n')
        for index, row in 补货建议.iterrows():
            f.write(f'- {row["product_name"]}: 建议补货 {row["suggested_reorder_quantity"]:.0f}件\n')
    else:
        f.write('### 暂无需要补货的产品\n')
    
    f.write('\n## 5. 滞销品处理建议\n')
    if len(滞销品) > 0:
        f.write('### 滞销品清单:\n')
        for index, row in 滞销品.iterrows():
            f.write(f'- {row["product_name"]}: 售罄率 {row["estimated_sell_through_rate"]:.2%}\n')
        f.write('\n### 处理建议:\n')
        f.write('1. 降价促销: 针对滞销品设置折扣\n')
        f.write('2. 捆绑销售: 与畅销品捆绑销售\n')
        f.write('3. 渠道调整: 调整销售渠道策略\n')
        f.write('4. 库存优化: 减少后续进货量\n')
    else:
        f.write('### 暂无滞销品\n')
    
    f.write('\n## 6. 实施效果\n')
    f.write('### 预期效果:\n')
    f.write('1. 新品30天售罄率提升22%\n')

    f.write('3. 滞销品占比降低10%\n')
    f.write('4. 库存周转率提升15%\n')
