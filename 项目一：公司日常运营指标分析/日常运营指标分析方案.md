# 项目一：公司日常运营指标分析方案

## 1. 项目概述

本项目旨在通过分析店铺的日常运营指标，探索用户行为模式，发现销售环节的瓶颈，并提出优化建议，以提高整体销售转化率。

## 2. 数据获取

### 数据库结构

#### `user_behavior`表(用户行为表)
| 字段名 | 数据类型 | 描述 |
|-------|---------|------|
| `id` | INT | 行为ID |
| `user_id` | INT | 用户ID |
| `action` | VARCHAR(50) | 行为类型（view(查看), add_to_cart(添加购物车), place_order(下单订单), pay(支付)） |
| `product_id` | INT | 产品ID |
| `create_time` | DATETIME | 行为时间 |

#### `orders`表(订单表)
| 字段名 | 数据类型 | 描述 |
|-------|---------|------|
| `order_id` | INT | 订单ID |
| `user_id` | INT | 用户ID |
| `amount` | DECIMAL(10,2) | 订单金额 |
| `order_status` | VARCHAR(50) | 订单状态（unpaid(未支付), paid(已支付), shipped(已发货), completed(已完成), cancelled(已取消)） |
| `create_time` | DATETIME | 订单创建时间 |

### MySQL查询语句

```sql
-- 1. 日PV和UV分析
SELECT 
    date(create_time) as date,
    COUNT(*) as pv,
    COUNT(DISTINCT user_id) as uv
FROM 
    user_behavior
WHERE 
--    create_time BETWEEN '2025-01-01' AND '2025-12-31'
      create_time BETWEEN '2025-01-01' AND '2025-12-31 23:59:59'
      -- 上面那个语句默认是and '2025-12-31 0:00:00',是从31号零点开始,后面包括的才是全部31号的时间
GROUP BY 
    date(create_time)
ORDER BY 
    date(create_time);

-- 2. 付费率分析
SELECT 
    DATE(create_time) AS dt,
    COUNT(DISTINCT user_id) AS total_users,
    COUNT(DISTINCT IF(action = 'pay', user_id, NULL)) AS paid_users,
    ROUND(
        COUNT(DISTINCT IF(action = 'pay', user_id, NULL)) * 1.0 
        / COUNT(DISTINCT user_id), 
        4
    ) AS payment_rate
FROM 
    user_behavior
WHERE 
    create_time >= '2025-01-01' 
    AND create_time < '2026-01-01'
GROUP BY 
    DATE(create_time)
ORDER BY 
    dt;

-- 3. 复购行为分析
SELECT 
    user_id,
    COUNT(DISTINCT order_id) as order_count,
    SUM(amount) as total_amount
FROM 
    orders
WHERE 
    order_status = 'paid'
    AND create_time BETWEEN '2025-01-01' AND '2025-12-31'
GROUP BY 
    user_id
HAVING 
    COUNT(DISTINCT order_id) >= 2;

-- 4. 漏斗流失分析
SELECT 
    action AS step,
    COUNT(DISTINCT user_id) AS user_count,
    ROUND(
        COUNT(DISTINCT user_id) * 1.0 
        / (SELECT COUNT(DISTINCT user_id) FROM user_behavior WHERE action = 'view' AND create_time >= '2025-01-01' AND create_time < '2026-01-01'), 
        4
    ) AS conversion_rate
FROM 
    user_behavior
WHERE 
    create_time >= '2025-01-01' 
    AND create_time < '2026-01-01'
    AND action IN ('view', 'add_to_cart', 'place_order', 'pay')
GROUP BY 
    action
ORDER BY 
    FIELD(action, 'view', 'add_to_cart', 'place_order', 'pay');

-- 5. RFM分析
SELECT 
    user_id,
    DATEDIFF(NOW(), MAX(create_time)) as recency,
    COUNT(DISTINCT order_id) as frequency,
    SUM(amount) as monetary
FROM 
    orders
WHERE 
    order_status = 'paid'
    AND create_time BETWEEN '2025-01-01' AND '2025-12-31'
GROUP BY 
    user_id;
```

## 3. 数据分析

### Python代码

```python
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
import pymysql

# 连接数据库
conn = pymysql.connect(
    host='localhost',
    user='root',
    password='password',
    database='ecommerce'
)

# 1. 日PV和UV分析
pv_uv_query = """
SELECT 
    date(create_time) as date,
    COUNT(*) as pv,
    COUNT(DISTINCT user_id) as uv
FROM 
    user_behavior
WHERE 
    create_time BETWEEN '2025-01-01' AND '2025-12-31'
GROUP BY 
    date(create_time)
ORDER BY 
    date(create_time);
"""
pv_uv_df = pd.read_sql(pv_uv_query, conn)

# 2. 付费率分析
payment_rate_query = """
SELECT 
    date(create_time) as date,
    COUNT(DISTINCT user_id) as total_users,
    COUNT(DISTINCT CASE WHEN order_status = 'paid' THEN user_id END) as paid_users,
    COUNT(DISTINCT CASE WHEN order_status = 'paid' THEN user_id END) / COUNT(DISTINCT user_id) as payment_rate
FROM 
    user_behavior
WHERE 
    create_time BETWEEN '2025-01-01' AND '2025-12-31'
GROUP BY 
    date(create_time)
ORDER BY 
    date(create_time);
"""
payment_rate_df = pd.read_sql(payment_rate_query, conn)

# 3. 复购行为分析
repurchase_query = """
SELECT 
    user_id,
    COUNT(DISTINCT order_id) as order_count,
    SUM(amount) as total_amount
FROM 
    orders
WHERE 
    order_status = 'paid'
    AND create_time BETWEEN '2025-01-01' AND '2025-12-31'
GROUP BY 
    user_id
HAVING 
    COUNT(DISTINCT order_id) >= 2;
"""
repurchase_df = pd.read_sql(repurchase_query, conn)

# 4. 漏斗流失分析
funnel_query = """
SELECT 
    step,
    COUNT(*) as user_count,
    COUNT(*) / (SELECT COUNT(*) FROM user_behavior WHERE action = 'view' AND create_time BETWEEN '2025-01-01' AND '2025-12-31') as conversion_rate
FROM (
    SELECT 
        user_id,
        MAX(CASE WHEN action = 'view' THEN 1 ELSE 0 END) as view,
        MAX(CASE WHEN action = 'add_to_cart' THEN 1 ELSE 0 END) as add_to_cart,
        MAX(CASE WHEN action = 'place_order' THEN 1 ELSE 0 END) as place_order,
        MAX(CASE WHEN action = 'pay' THEN 1 ELSE 0 END) as pay
    FROM 
        user_behavior
    WHERE 
        create_time BETWEEN '2025-01-01' AND '2025-12-31'
    GROUP BY 
        user_id
) as funnel
UNPIVOT (
    is_completed FOR step IN (view, add_to_cart, place_order, pay)
) as unpivoted
WHERE 
    is_completed = 1
GROUP BY 
    step
ORDER BY 
    FIELD(step, 'view', 'add_to_cart', 'place_order', 'pay');
"""
funnel_df = pd.read_sql(funnel_query, conn)

# 5. RFM分析
rfm_query = """
SELECT 
    user_id,
    DATEDIFF(NOW(), MAX(create_time)) as recency,
    COUNT(DISTINCT order_id) as frequency,
    SUM(amount) as monetary
FROM 
    orders
WHERE 
    order_status = 'paid'
    AND create_time BETWEEN '2025-01-01' AND '2025-12-31'
GROUP BY 
    user_id;
"""
rfm_df = pd.read_sql(rfm_query, conn)

# 计算RFM得分
rfm_df['r_score'] = pd.qcut(rfm_df['recency'], 5, labels=[5, 4, 3, 2, 1])
rfm_df['f_score'] = pd.qcut(rfm_df['frequency'], 5, labels=[1, 2, 3, 4, 5])
rfm_df['m_score'] = pd.qcut(rfm_df['monetary'], 5, labels=[1, 2, 3, 4, 5])

# 计算总得分
rfm_df['rfm_score'] = rfm_df['r_score'].astype(int) + rfm_df['f_score'].astype(int) + rfm_df['m_score'].astype(int)

# 客户分层
def rfm_segment(row):
    if row['rfm_score'] >= 12:
        return '高价值客户'
    elif row['rfm_score'] >= 9:
        return '中高价值客户'
    elif row['rfm_score'] >= 6:
        return '中价值客户'
    else:
        return '低价值客户'

rfm_df['segment'] = rfm_df.apply(rfm_segment, axis=1)

# 数据可视化
plt.figure(figsize=(12, 8))

# 1. PV和UV趋势
plt.subplot(2, 2, 1)
plt.plot(pv_uv_df['date'], pv_uv_df['pv'], label='PV')
plt.plot(pv_uv_df['date'], pv_uv_df['uv'], label='UV')
plt.title('日PV和UV趋势')
plt.xlabel('日期')
plt.ylabel('数量')
plt.legend()

# 2. 付费率趋势
plt.subplot(2, 2, 2)
plt.plot(payment_rate_df['date'], payment_rate_df['payment_rate'])
plt.title('付费率趋势')
plt.xlabel('日期')
plt.ylabel('付费率')

# 3. 漏斗分析
plt.subplot(2, 2, 3)
plt.bar(funnel_df['step'], funnel_df['user_count'])
plt.title('漏斗分析')
plt.xlabel('步骤')
plt.ylabel('用户数')

# 4. 客户分层
plt.subplot(2, 2, 4)
segment_counts = rfm_df['segment'].value_counts()
plt.pie(segment_counts, labels=segment_counts.index, autopct='%1.1f%%')
plt.title('客户分层')

plt.tight_layout()
plt.savefig('日常运营指标分析.png')

# 关闭数据库连接
conn.close()
