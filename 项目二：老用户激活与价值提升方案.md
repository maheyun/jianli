# 项目二：老用户激活与价值提升方案

## 1. 项目概述

本项目旨在通过构建RFM-G用户分层模型，设计差异化运营策略，扭转老用户GMV占比下滑趋势，提升其复购率与客单价。

### MySQL查询语句

```sql
-- 1. 老用户行为数据
SELECT 
    u.user_id,
    u.registration_date,
    COUNT(DISTINCT o.order_id) as order_count,
    SUM(o.amount) as total_amount,
    MAX(o.create_time) as last_purchase_date,
    COUNT(DISTINCT CASE WHEN o.create_time BETWEEN DATE_SUB(NOW(), INTERVAL 30 DAY) AND NOW() THEN o.order_id END) as recent_30d_orders,
    COUNT(DISTINCT CASE WHEN p.category_id = 1 THEN o.order_id END) as category_1_orders, -- 袜子品类
    COUNT(DISTINCT CASE WHEN p.category_id = 2 THEN o.order_id END) as category_2_orders -- 服装品类
FROM 
    users u
LEFT JOIN 
    orders o ON u.user_id = o.user_id
LEFT JOIN 
    order_items oi ON o.order_id = oi.order_id
LEFT JOIN 
    products p ON oi.product_id = p.product_id
WHERE 
    u.registration_date < DATE_SUB(NOW(), INTERVAL 6 MONTH) -- 老用户定义：注册超过6个月
    AND o.order_status = 'paid'
GROUP BY 
    u.user_id, u.registration_date;

-- 2. 活动参与数据
SELECT 
    user_id,
    COUNT(DISTINCT activity_id) as activity_count,
    SUM(CASE WHEN is_participated = 1 THEN 1 ELSE 0 END) as participation_count
FROM 
    user_activities
WHERE 
    activity_date BETWEEN DATE_SUB(NOW(), INTERVAL 2 YEAR) AND NOW()
GROUP BY 
    user_id;
```

## 3. 数据分析

### Python代码

```python
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
import pymysql

# 连接数据库
conn = pymysql.connect(
    host='localhost',
    user='root',
    password='password',
    database='ecommerce'
)

# 1. 老用户行为数据
user_behavior_query = """
SELECT 
    u.user_id,
    u.registration_date,
    COUNT(DISTINCT o.order_id) as order_count,
    SUM(o.amount) as total_amount,
    MAX(o.create_time) as last_purchase_date,
    COUNT(DISTINCT CASE WHEN o.create_time BETWEEN DATE_SUB(NOW(), INTERVAL 30 DAY) AND NOW() THEN o.order_id END) as recent_30d_orders,
    COUNT(DISTINCT CASE WHEN p.category_id = 1 THEN o.order_id END) as category_1_orders,
    COUNT(DISTINCT CASE WHEN p.category_id = 2 THEN o.order_id END) as category_2_orders
FROM 
    users u
LEFT JOIN 
    orders o ON u.user_id = o.user_id
LEFT JOIN 
    order_items oi ON o.order_id = oi.order_id
LEFT JOIN 
    products p ON oi.product_id = p.product_id
WHERE 
    u.registration_date < DATE_SUB(NOW(), INTERVAL 6 MONTH)
    AND o.order_status = 'paid'
GROUP BY 
    u.user_id, u.registration_date;
"""
user_behavior_df = pd.read_sql(user_behavior_query, conn)

# 2. 活动参与数据
activity_query = """
SELECT 
    user_id,
    COUNT(DISTINCT activity_id) as activity_count,
    SUM(CASE WHEN is_participated = 1 THEN 1 ELSE 0 END) as participation_count
FROM 
    user_activities
WHERE 
    activity_date BETWEEN DATE_SUB(NOW(), INTERVAL 2 YEAR) AND NOW()
GROUP BY 
    user_id;
"""
activity_df = pd.read_sql(activity_query, conn)

# 合并数据
df = pd.merge(user_behavior_df, activity_df, on='user_id', how='left')
df = df.fillna(0)

# 计算RFM-G指标
df['recency'] = (pd.Timestamp.now() - pd.to_datetime(df['last_purchase_date'])).dt.days
df['frequency'] = df['order_count']
df['monetary'] = df['total_amount']
df['growth'] = df['category_2_orders'] / (df['category_1_orders'] + 1)  # 品类拓展成长性

# 计算RFM-G得分
df['r_score'] = pd.qcut(df['recency'], 5, labels=[5, 4, 3, 2, 1])
df['f_score'] = pd.qcut(df['frequency'], 5, labels=[1, 2, 3, 4, 5])
df['m_score'] = pd.qcut(df['monetary'], 5, labels=[1, 2, 3, 4, 5])
df['g_score'] = pd.qcut(df['growth'], 5, labels=[1, 2, 3, 4, 5])

# 计算总得分
df['rfmg_score'] = df['r_score'].astype(int) + df['f_score'].astype(int) + df['m_score'].astype(int) + df['g_score'].astype(int)

# 用户分层
def rfmg_segment(row):
    if row['rfmg_score'] >= 16:
        return '高价值深耕用户'
    elif row['rfmg_score'] >= 12:
        return '高潜唤醒用户'
    elif row['rfmg_score'] >= 8:
        return '成长型用户'
    else:
        return '流失风险用户'

df['segment'] = df.apply(rfmg_segment, axis=1)

# 分析各分层用户特征
segment_analysis = df.groupby('segment').agg({
    'order_count': 'mean',
    'total_amount': 'mean',
    'recency': 'mean',
    'growth': 'mean',
    'activity_count': 'mean',
    'participation_count': 'mean'
}).round(2)

# 数据可视化
plt.figure(figsize=(15, 10))

# 1. 用户分层层级
plt.subplot(2, 2, 1)
segment_counts = df['segment'].value_counts()
plt.pie(segment_counts, labels=segment_counts.index, autopct='%1.1f%%')
plt.title('用户分层分布')

# 2. 各分层用户平均订单数
plt.subplot(2, 2, 2)
plt.bar(segment_analysis.index, segment_analysis['order_count'])
plt.title('各分层用户平均订单数')
plt.ylabel('平均订单数')

# 3. 各分层用户平均消费金额
plt.subplot(2, 2, 3)
plt.bar(segment_analysis.index, segment_analysis['total_amount'])
plt.title('各分层用户平均消费金额')
plt.ylabel('平均消费金额')

# 4. 各分层用户平均成长性
plt.subplot(2, 2, 4)
plt.bar(segment_analysis.index, segment_analysis['growth'])
plt.title('各分层用户平均品类拓展成长性')
plt.ylabel('成长性指数')

plt.tight_layout()
plt.savefig('老用户激活与价值提升分析.png')

# 关闭数据库连接
conn.close()
